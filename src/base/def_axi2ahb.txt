INCLUDE   def_axi2ahb_static.txt

SWAP.USER      DATA_BITS   32              ##AXI and AHB data bits
SWAP.USER      SIZE_BITS   2               ##AXI size bits
SWAP.USER      ID_BITS     4               ##AXI ID bits
